abstract class View{
+View():
+{abstract} mostrarMenu(): void
+{abstract} mostrarNivel(int nivel): void
+{abstract} mostrarMapa(Tablero& paradero): void
+{abstract} pantallaDerrota() : void
+{abstract} pantallaVictoria(): void
+virtual ~View():

}

abstract class ViewSFML{
+ViewSFML()
+mostrarMenu(): void override
+mostrarNivel(int nivel): void override
+mostrarMapa(Tablero& paradero): void override
+pantallaDerrota() : void override
+pantallaVictoria(): void override
+~ViewSFML(): override
}

abstract class ViewConsola{
+ViewConsola()

+mostrarNivel(int nivel): void override
+mostrarMapa(Tablero& paradero): void override
+pantallaDerrota() : void override

}
